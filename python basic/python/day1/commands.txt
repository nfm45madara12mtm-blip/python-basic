# GIT: source control, tracks file and folder, enable collaboration(branching,merge)
# compare the previous state and current state of the folder
# initialize local git repo
git init

# add file to the git
git add file_name          [#add single file]
git add .                  [# add multiple files]

# commit file to git
git commit -m "message"

# status check of this repo
git status

# see previous commits
git log

# define user identity
git config --global user.name "your_username"
git config --global user.email your_email

# Branch:
#check branchs
git Branch

# create new branch
git branch branch_name

# rename branch name
git branch -M "new_branch_name"

# merge branch
git merge branch_name

# switch branch
git switch branch_name
git checkout branch_name

# merge the files
git merge branch_name
